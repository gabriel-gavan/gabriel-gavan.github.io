Classes and Objects in C++

Classes group data and behavior. Use constructors to guarantee valid state and mark methods const when not mutating.

class User {
  std::string name;
  int points {0};
public:
  explicit User(std::string n) : name(std::move(n)) {}
  void add_points(int p){ if(p > 0) points += p; }
  int get_points() const { return points; }
  const std::string& get_name() const { return name; }
  void greet() const { std::cout << "Hello, " << name; }
};

Rule of zero/five

// If you manage no resources, let compiler generate special members (rule of zero)
// If you manage resources, define or =delete copy/move/dtor appropriately (rule of five)
struct Buffer {
  size_t n{}; int* data{};
  Buffer(size_t n): n(n), data(new int[n]{}) {}
  ~Buffer(){ delete[] data; }
  Buffer(const Buffer& other): n(other.n), data(new int[n]){ std::copy(other.data, other.data+n, data); }
  Buffer& operator=(const Buffer& o){ if(this!=&o){ Buffer tmp(o); std::swap(n,tmp.n); std::swap(data,tmp.data);} return *this; }
  Buffer(Buffer&& o) noexcept : n(o.n), data(o.data){ o.n=0; o.data=nullptr; }
  Buffer& operator=(Buffer&& o) noexcept { if(this!=&o){ delete[] data; n=o.n; data=o.data; o.n=0; o.data=nullptr;} return *this; }
};

Using the class

int main(){

}
