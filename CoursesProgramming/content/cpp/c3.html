Operators and Control Flow

Write clear branches and loops; avoid deep nesting. Prefer early returns and modern C++ features.

Conditionals and switch

int score = 60;
if (score >= 50) {
  std::cout << "Passed";
} else {
  std::cout << "Try again";
}

enum class Day { Mon, Tue, Sat, Sun };
Day day = Day::Mon;
switch (day) {
  case Day::Mon: case Day::Tue: std::cout << "Work"; break;
  case Day::Sat: case Day::Sun: std::cout << "Rest"; break;
}

Loops and range‑for

for (int i = 0; i < 3; ++i) { std::cout << i; }
std::vector v{1,2,3};
for (int n : v) { std::cout << n; } // range-based for

if with initializer (C++17)

std::map m{{"a",1}};
if (auto it = m.find("a"); it != m.end()) {
  std::cout << it->second;
}
