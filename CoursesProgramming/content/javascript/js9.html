Fetch API and Error Handling

Fetch provides a promise‑based interface for HTTP requests. Always handle non‑200s and timeouts.

GET/POST with headers

async function getUser(id, { signal } = {}){
  const res = await fetch(`/users/${id}`, { signal, headers: { 'accept': 'application/json' } });
  if(!res.ok) throw new Error(`HTTP ${res.status}`);
  return res.json();
}

async function createUser(data, { signal } = {}){
  const res = await fetch('/users', {
    method: 'POST',
    headers: { 'content-type': 'application/json' },
    body: JSON.stringify(data),
    signal,
  });
  if(!res.ok) throw new Error(`HTTP ${res.status}`);
  return res.json();
}

Timeout with AbortController

function fetchWithTimeout(input, init = {}, ms = 5000){
  const ctrl = new AbortController();
  const t = setTimeout(() => ctrl.abort(), ms);
  return fetch(input, { ...init, signal: ctrl.signal })
    .finally(() => clearTimeout(t));
}
